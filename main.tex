            \State text

            \State

            \State text

            \State text

            \State text  
            \EndIf
            \State
            
            \For{condition}   
            \State text
            \EndFor
            \State
                
            \ForAll {condition}
            \State text
            \EndFor
            \State

            \State text 
            \EndWhile
            \State
                
            \Repeat 
            \State text
            \Until {condition}
            \State
                
            \Loop 
            \State text
            \EndLoop
            \State
            
            \State \Return{text}
        \end{algorithmic}
